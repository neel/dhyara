63  VectorIt parse(VectorIt v_it, VectorIt v_end, StrIt a_it, StrIt a_end){
64  auto rest = std::distance(a_it, a_end);
65  std::stringstream stream;
66  if(rest > 0){
67  // value provided in head
68  std::copy(a_it, a_end, std::ostream_iterator<char>(stream));
69  read(stream);
70  return v_it+1;
71  }else{
72  // value provided in tail
73  if(v_it+1 != v_end) {
74  stream << *(++v_it);
75  read(stream);
76  return v_it+1;
77  }else{

79  return v_end;
80  }

102  if(!_option.empty()){
103  stream << _prefix << _option << "<" << _placeholder << ">";
104  chars_written += _prefix.size() + _option.size() + _placeholder.size() + 2;
105  }
106  if(!_option.empty() && !_name.empty()){
107  stream << ", ";
108  chars_written += 2;
109  }
110  if(!_name.empty()){
111  stream << _prefix << _prefix << _name << " " << "<" << _placeholder << ">";
112  chars_written += 2*_prefix.size() + _name.size() + 1 + _placeholder.size() + 2;
113  }
114  }else{
115  stream << "<" << _placeholder << ">";
116  chars_written += _placeholder.size() + 2;
117  }
118  if(chars_written < 35){
119  std::string space(35-chars_written, ' ');
120  stream << space;
121  }
122  stream << " " << _description;
