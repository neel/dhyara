24  template <typename InputIt>
25  InputIt parse(InputIt v_begin, InputIt v_end){
26  InputIt it = v_begin;
27  InputIt next = it;
28  while(it != v_end){
29  next = _chain._parse(it, v_end);
30  if(next == it){
31  next = _chain._positional(it);
32  it = (next == it) ? next +1 : next;
33  }else{
34  it = next;
35  }
36  }
37  return it;
38  }
39 
40  private:
41  chain<Head, Tail>& _chain;
42  };
