20 template<class ForwardIt1, class ForwardIt2>
21 ForwardIt1 search(ForwardIt1 first, ForwardIt1 last, ForwardIt2 s_first, ForwardIt2 s_last){
22  while (true) {
23  ForwardIt1 it = first;
24  for (ForwardIt2 s_it = s_first; ; ++it, ++s_it) {
25  if (s_it == s_last) return first;
26  if (it == last) return last;
27  if (!(*it == *s_it)) break;
28  }
29  ++first;
30  }
31 }

33 template <typename T, typename ArgT>
34 static auto match_param(const param<T>& p, const ArgT& arg) -> decltype(arg.begin()){
35  auto begin = std::begin(arg);
36  auto end = std::end(arg);
37  std::string name = p.name(), option = p.option();
38 
39  // search for the name that starts with double hyphens
40  auto it = name.empty() ? end : detail::search(begin, end, name.begin(), name.end());
41  if(it != end){
42  std::advance(it, name.size()-1);
43  }else{
44  // search for the option that starts with single hyphen
45  it = option.empty() ? end : detail::search(begin, end, option.begin(), option.end());
46  if(it != end){
47  std::advance(it, option.size()-1);
48  }
49  }
50  return it;
51 }
